    <script>
        const questions = [
            "Yere çöp atar mısın?",
            "Ana arter bir yolu düğün veya asker konvoyu ile keser misin?",
            "Ambulansa yol verir misin?",
            "Yılda en az bir tane kitap okur musun?",
            "Kendi görünüş tarzından farklı bir insanla karşılaşırsan müdahale eder misin?"
        ];

        let currentQuestion = 0;

        function loadQuestion() {
            if (currentQuestion < questions.length) {
                document.getElementById('questionContainer').innerHTML = `
                    <div class="question">${questions[currentQuestion]}</div>
                    <button class="button" onclick="handleAnswer(true)">Evet</button>
                    <button class="button" onclick="handleAnswer(false)">Hayır</button>
                `;
            } else {
                document.getElementById('questionContainer').style.display = 'none';
                document.getElementById('resultContainer').innerText = 'Testi tamamladınız!';
            }
        }

        function handleAnswer(answer) {
            if (!answer) {
                document.getElementById('resultContainer').innerText = 'Sen bir köylüsün, lütfen geldiğin köye geri dön.';
                document.getElementById('questionContainer').style.display = 'none';
            } else {
                currentQuestion++;
                loadQuestion();
            }
        }

        loadQuestion();
    </script>
